34 inline Result<uid_t> getuid(const Option<std::string>& user = None())
35 {
36  if (user.isNone()) {
37  return ::getuid();
38  }
39 
40  struct passwd passwd;
41  struct passwd* result = nullptr;
42 
43  int size = sysconf(_SC_GETPW_R_SIZE_MAX);
44  if (size == -1) {
45  // Initial value for buffer size.
46  size = 1024;
47  }
48 
49  while (true) {
50  char* buffer = new char[size];
51 
52  if (getpwnam_r(user.get().c_str(), &passwd, buffer, size, &result) == 0) {
53  // The usual interpretation of POSIX is that getpwnam_r will
54  // return 0 but set result == nullptr if the user is not found.
55  if (result == nullptr) {
56  delete[] buffer;
57  return None();
58  }
59 
60  uid_t uid = passwd.pw_uid;
61  delete[] buffer;
62  return uid;
63  } else {
64  // RHEL7 (and possibly other systems) will return non-zero and
65  // set one of the following errors for "The given name or uid
66  // was not found." See 'man getpwnam_r'. We only check for the
67  // errors explicitly listed, and do not consider the ellipsis.
68  if (errno == ENOENT ||
69  errno == ESRCH ||
70  errno == EBADF ||
71  errno == EPERM) {
72  delete[] buffer;
73  return None();
74  }
75 
76  if (errno != ERANGE) {
77  delete[] buffer;
78  return ErrnoError("Failed to get username information");
79  }
80  // getpwnam_r set ERANGE so try again with a larger buffer.
81  size *= 2;
82  delete[] buffer;
83  }
84  }
85 
86  UNREACHABLE();
87 }

166 inline Try<std::vector<gid_t>> getgrouplist(const std::string& user)
167 {
168  // TODO(jieyu): Consider adding a 'gid' parameter and avoid calling
169  // getgid here. In some cases, the primary gid might be known.
170  Result<gid_t> gid = os::getgid(user);
171  if (!gid.isSome()) {
172  return Error("Failed to get the gid of the user: " +
173  (gid.isError() ? gid.error() : "group not found"));
174  }
175 
176 #ifdef __APPLE__
177  // TODO(gilbert): Instead of setting 'ngroups' as a large value,
178  // we should figure out a way to probe 'ngroups' on OS X. Currently
179  // neither '_SC_NGROUPS_MAX' nor 'NGROUPS_MAX' is appropriate,
180  // because both are fixed as 16 on Darwin kernel, which is the
181  // cache size.
182  int ngroups = 65536;
183  int gids[ngroups];
184 #else
185  int ngroups = NGROUPS_MAX;
186  gid_t gids[ngroups];
187 #endif
188  if (::getgrouplist(user.c_str(), gid.get(), gids, &ngroups) == -1) {
189  return ErrnoError();
190  }
191 
192  return std::vector<gid_t>(gids, gids + ngroups);
193 }

196 inline Try<Nothing> setgroups(
197  const std::vector<gid_t>& gids,
198  const Option<uid_t>& uid = None())
199 {
200  int ngroups = static_cast<int>(gids.size());
201  gid_t _gids[ngroups];
202 
203  for (int i = 0; i < ngroups; i++) {
204  _gids[i] = gids[i];
205  }
206 
207 #ifdef __APPLE__
208  // Cannot simply call 'setgroups' here because it only updates
209  // the list of groups in kernel cache, but not the ones in
210  // opendirectoryd. Darwin kernel caches part of the groups in
211  // kernel, and the rest in opendirectoryd.
212  // For more detail please see:
213  // https://github.com/practicalswift/osx/blob/master/src/samba/patches/support-darwin-initgroups-syscall // NOLINT
214  int maxgroups = sysconf(_SC_NGROUPS_MAX);
215  if (maxgroups == -1) {
216  return Error("Failed to get sysconf(_SC_NGROUPS_MAX)");
217  }
218 
219  if (ngroups > maxgroups) {
220  ngroups = maxgroups;
221  }
222 
223  if (uid.isNone()) {
224  return Error(
225  "The uid of the user who is associated with the group "
226  "list we are setting is missing");
227  }
228 
229  // NOTE: By default, the maxgroups on Darwin kernel is fixed
230  // as 16. If we have more than 16 gids to set for a specific
231  // user, then SYS_initgroups would send up to 16 of them to
232  // kernel cache, while the rest would still be performed
233  // correctly by the kernel (asking Directory Service to resolve
234  // the groups membership).
235  if (::syscall(SYS_initgroups, ngroups, _gids, uid.get()) == -1) {
236  return ErrnoError();
237  }
238 #else
239  if (::setgroups(ngroups, _gids) == -1) {
240  return ErrnoError();
241  }
242 #endif
243 
244  return Nothing();
245 }

290 inline Try<Nothing> su(const std::string& user)
291 {
292  Result<gid_t> gid = os::getgid(user);
293  if (gid.isError() || gid.isNone()) {
294  return Error("Failed to getgid: " +
295  (gid.isError() ? gid.error() : "unknown user"));
296  } else if (::setgid(gid.get())) {
297  return ErrnoError("Failed to set gid");
298  }
299 
300  // Set the supplementary group list. We ignore EPERM because
301  // performing a no-op call (switching to same group) still
302  // requires being privileged, unlike 'setgid' and 'setuid'.
303  if (::initgroups(user.c_str(), gid.get()) == -1 && errno != EPERM) {
304  return ErrnoError("Failed to set supplementary groups");
305  }
306 
307  Result<uid_t> uid = os::getuid(user);
308  if (uid.isError() || uid.isNone()) {
309  return Error("Failed to getuid: " +
310  (uid.isError() ? uid.error() : "unknown user"));
311  } else if (::setuid(uid.get())) {
312  return ErrnoError("Failed to setuid");
313  }
314 
315  return Nothing();
316 }
